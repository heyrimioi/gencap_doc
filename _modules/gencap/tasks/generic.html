@task_manager.register("GENCAP-DatasetReproductionTask", version="1.0")
class DatasetReproductionTask(Task):
    def __init__(self):
        super().__init__()
        self.unique_id = "GENCAP-DatasetReproductionTask"
        self.version = "1.0"

        # Task metadata
        self.application_domain = "Generic"
        self.application = ["validation", "reproducibility"]
        self.developer_names = "GENCAP Dev Team"
        self.developer_organization = "Inria"

        # Description
        self.description = "This task reproduces a derived dataset by reapplying the same pipeline and parameters originally used to generate it. It evaluates the reproducibility and integrity of the dataset creation process."
        self.parameter_info = {"dataset_dir":"(str) Path to reproduce which have data_source.json",
                              "keep_intermediate_datasets":"(bool) parameter of the pipeline runtime to keep the intermediate data or not. default:True",
                              "eval":"(bool) if eval is True Bit-wise comparison between reproducted derivatives and refered dataset after reproduction task. default:True",
                              "n_level":"(int) set the number of level to conduct recursion. if it is -1 it is executing the reproduction from the cohort. default:-1"}

        self.input_file_location_fields = ["Original Path"]
        self.output_file_location_fields = ["Reproduced Path"]

        self.mandatory_input_fields = ["Execution Order","Level","Original Path"]
        self.optional_input_fields = []
        self.additional_output_indexing_fields = []
        self.mandatory_output_fields = ["Execution Order","Level","Original Path","Reproduced Dataset","Reproduced Path","Execution Time(min)"]
        self.optional_output_fields = []

    def _run_task(self,
                 input_dataframe: pd.DataFrame=None,
                 output_dataframe: pd.DataFrame=None,
                 error_dataframe: pd.DataFrame=None,
                 base_indexing_fields: list=None,
                 parameter_dict: dict=None,
                  pathname_policy_dict:dict=None,
                 debug_mode: bool = False):

        import json
        import os
        import time
        import copy

        import pandas as pd

        from gencap import Cohort, Pipeline
        from gencap.utils.helpers import (
            find_dataset_path,
            generate_unique_name_with_prefix,
        )

        if input_dataframe is None:
            input_dataframe = self.input_dataframe
        if output_dataframe is None:
            output_dataframe = self.output_dataframe
        if error_dataframe is None:
            error_dataframe = self.error_dataframe
        if base_indexing_fields is None:
            base_indexing_fields = self.base_indexing_fields
        if parameter_dict is None and self.parameter_dict is not None:
            parameter_dict = self.parameter_dict
        if pathname_policy_dict is None and self.pathname_policy_dict is not None:
            pathname_policy_dict = self.pathname_policy_dict

        self.logger.info(f"[{self.unique_id}] has launched...")

        # Get the path to the input dataset path from the parameter
        new_datasetDir_path = self.output_dir.split("derivatives")[0]

        if "prefix" not in parameter_dict.keys():
            parameter_dict["prefix"] = self.dataset_unique_id

        # make name_mapper
        # distinguish intermediate folder
        inter_data = input_dataframe[input_dataframe["Original Path"].str.contains("intermediate_data", na=False)].index.tolist()
        reproduce_name_mapper = {}
        for data in input_dataframe.index.tolist():
            if data in inter_data:
                prefix = f"{parameter_dict['prefix']}-i"
            else:
                prefix = parameter_dict["prefix"]

            reproduce_name_mapper[data] = generate_unique_name_with_prefix(new_datasetDir_path, data, prefix, logger=self.logger)
        output_dataframe = pd.DataFrame()
        error_dataframe = pd.DataFrame()

        # Iterate the dataframe by row
        for i, row in input_dataframe.reset_index().iterrows():
            try:
                start_time = time.time()
                reproduced_dataset = reproduce_name_mapper[row["Original Data"]]
                pipe_path = row["Original Path"]

                # Restruct Cohort
                self.logger.info(f"Reconstructing cohort from data_source.json in {pipe_path}...")
                data_source = json.load(open(os.path.join(pipe_path,"data_source.json"), "r"))

                if i ==0:
                    cohort = Cohort(datasetDir=data_source["cohortArguments"]["datasetDir"], 
                                    datasetList=data_source["cohortArguments"]["datasetList"],
                                    tagList=data_source["cohortArguments"]["tagList"],
                                    fieldFormatDict=data_source["cohortArguments"]["fieldFormatDict"],
                                    additionalFieldDict=data_source["cohortArguments"]["additionalFieldDict"],
                                    replaceValueDict=data_source["cohortArguments"]["replaceValueDict"],
                                    changeFieldNamesDict=data_source["cohortArguments"]["changeFieldNamesDict"],
                                    excludeFieldList=data_source["cohortArguments"]["excludeFieldList"],
                                    mandatoryFieldList=data_source["cohortArguments"]["mandatoryFieldList"],
                                    optionalFieldList=data_source["cohortArguments"]["optionalFieldList"],
                                    excludeEntriesList=data_source["cohortArguments"]["excludeEntriesList"],
                                    queryString=data_source["cohortArguments"]["queryString"])

                else:
                    cohort = Cohort(datasetDir=new_datasetDir_path, 
                                    datasetList=[reproduce_name_mapper.get(item, item) for item in data_source["cohortArguments"]["datasetList"]],
                                    tagList=data_source["cohortArguments"]["tagList"],
                                    fieldFormatDict=data_source["cohortArguments"]["fieldFormatDict"],
                                    additionalFieldDict=data_source["cohortArguments"]["additionalFieldDict"],
                                    replaceValueDict=data_source["cohortArguments"]["replaceValueDict"],
                                    changeFieldNamesDict=data_source["cohortArguments"]["changeFieldNamesDict"],
                                    excludeFieldList=data_source["cohortArguments"]["excludeFieldList"],
                                    mandatoryFieldList=data_source["cohortArguments"]["mandatoryFieldList"],
                                    optionalFieldList=data_source["cohortArguments"]["optionalFieldList"],
                                    excludeEntriesList=data_source["cohortArguments"]["excludeEntriesList"],
                                    queryString=data_source["cohortArguments"]["queryString"])

                # Restruct Pipeline
                self.logger.info(f"Reconstructing pipeline from data_source.json in {pipe_path}...")
                pipeline = Pipeline(dataset_filename = pipe_path)

                # Assign the inherited runtime arguments from the pipeline to launch run
                tag = copy.deepcopy(self.additional_tag_list)
                tag.append(f"Reproduced from {parameter_dict['dataset_dir']}")
                # Run pipeline
                pipeline.run(
                    cohort=cohort,
                    output_dir=os.path.join(self.output_dir, self.dataset_unique_id),
                    dataset_id=reproduced_dataset,
                    input_dict=data_source["pipelineArguments"]["input_dict"],
                    additional_tag_list=tag,
                    force_dir_replacement=self.force_dir_replacement,
                    keep_intermediate_datasets=parameter_dict["keep_intermediate_datasets"],
                    use_multithreading=self.use_multithreading,
                    nb_max_threads=self.nb_max_threads,
                    pathname_policy_id=self.pathname_policy_id,
                    pathname_policy_dict=self.pathname_policy_dict,
                    debug_mode=self.debug_mode
                )
                end_time = time.time()

                row["Reproduced Dataset"] = reproduced_dataset
                row["Reproduced Path"] = find_dataset_path(datasetName=reproduced_dataset,databaseDir=new_datasetDir_path)
                row["Execution Time(min)"] = round((end_time - start_time)/60, 2)
                row_output = pd.DataFrame([row])
                output_dataframe = pd.concat([output_dataframe,  row_output], ignore_index=False)

            except Exception as e:
                # Log any exceptions encountered during processing
                error_row = {"Base Indexing Value": str(row["Original Data"]),
                            "Additional Indexing Value": "",
                            "Error Code": "ProcessingFail",
                            "Description": str(e),
                            "Values": str(row)}
                error_dataframe = pd.concat([error_dataframe, pd.DataFrame([error_row])], ignore_index=True)
                self.logger.error(str(e))
                import traceback
                tb_str = traceback.format_exc()
                self.logger.error("Traceback message:")
                self.logger.error(tb_str)

        return output_dataframe, error_dataframe
    def _run_quality_control(self,HTMLReport:"HTMLReport",
                             nicePlots=True,
                            offline_mode=True):
        from gencap.utils.helpers import (
            build_dependency_graph_dataset,
            compare_datasets,
            visualize_graph_dataset,
        )
        hrQC = HTMLReport
        hrQC.addSection("Data Source Tree")
        hrQC.addContent(f"<p>Reproduced Path: {self.root_dir}</p>")
        G_orig = build_dependency_graph_dataset(dataset_dir=self.parameter_dict["dataset_dir"],n_level=self.parameter_dict["n_level"],logger=self.logger)
        fig_orig = visualize_graph_dataset(G_orig,return_fig=True,nicePlots=nicePlots,show=False,logger=self.logger)

        G_re = build_dependency_graph_dataset(dataset_dir=self.output_dataframe.iloc[-1]["Reproduced Path"],n_level=self.parameter_dict["n_level"],logger=self.logger)
        fig_re = visualize_graph_dataset(G_re,return_fig=True,nicePlots=nicePlots,show=False,logger=self.logger)

        hrQC.addFigurePair(fig_orig, fig_re, "Original Data Source Tree", "Reproduced Data Source Tree")

        if self.parameter_dict["eval"]:
            hrQC.addSection("Output Evaluation")
            total_result_df = pd.DataFrame()
            total_error_df = pd.DataFrame()
            for index, row in self.output_dataframe.iterrows():
                original_dir = row["Original Path"]
                reproduced_dir = row["Reproduced Path"]
                valid_code, result_df, error_df = compare_datasets(original_dir, reproduced_dir,logger=self.logger)
                total_result_df = pd.concat([total_result_df,result_df], ignore_index=True)
                total_error_df = pd.concat([total_error_df,error_df], ignore_index=True)

                if 1 in valid_code:
                    hrQC.addContent(
                        f"[{index}]: Indexing fields mismatch. Original Path: {row['Original Path']}, Reproduced Path: {row['Reproduced Path']}</p>"
                    )
                if 2 in valid_code:
                    hrQC.addContent(
                        f"[{index}]: Column fields mismatch. Original Path: {row['Original Path']}, Reproduced Path: {row['Reproduced Path']}</p>"
                    )
                if 3 in valid_code:
                    hrQC.addContent(
                        f"[{index}]: DataFrames are different in table contents. Original Path: {row['Original Path']}, Reproduced Path: {row['Reproduced Path']}</p>"
                    )
                if 4 in valid_code:
                    hrQC.addContent(
                        f"[{index}]: Some of files are different. Original Path: {row['Original Path']}, Reproduced Path: {row['Reproduced Path']}</p>"
                    )
                    hrQC.addDataFrame(total_error_df, name="Error List: ", hide=True)
                if len(valid_code) ==0:
                    hrQC.addContent(f"[{index}]: Reproduced dataset is identical with original dataset</p>")
        return hrQC

@task_manager.register("GENCAP-CohortSavingTask", version="1.0")
class CohortSavingTask(Task):
    def __init__(self):
        super().__init__()
        # Unique ID and version
        self.unique_id = "GENCAP-CohortSavingTask"
        self.version = "1.0"
        # Catalogue metadata
        self.application_domain = "Generic"
        self.application = ["utility", "saving"]
        self.developer_names = "GENCAP Dev Team"
        self.developer_organization = "Inria"
        # Human-readable description
        self.description = "This task saves a cohort into a derived dataset."
        # No parameters needed for this task
        self.parameter_info = {}

    def _run_task(
        self,
        input_dataframe: pd.DataFrame = None,
        output_dataframe: pd.DataFrame = None,
        error_dataframe: pd.DataFrame = None,
        base_indexing_fields: list = None,
        parameter_dict: dict = None,
        pathname_policy_dict: dict = None,
        debug_mode: bool = False
    ):
        """
        Full-batch processing: simply copy the cohort DataFrame to output,
        then let Task._finalize handle saving.
        """
        if self.cohort is None:
            raise AttributeError("Cohort was not assigned to the runtime arguments")
        
        # update input_fields and output_fields by cohort
        self.base_indexing_fields = self.cohort.get_indexing_fields()
        self.mandatory_input_fields = self.cohort.mandatoryFieldList
        self.optional_input_fields = self.cohort.optionalFieldList
        self.input_file_location_fields = self.cohort.get_file_location_fields()       

        self.mandatory_output_fields = self.cohort.mandatoryFieldList
        self.optional_output_fields = self.cohort.optionalFieldList
        self.output_file_location_fields = self.cohort.get_file_location_fields()      
        
        # re-save with updated variables
        self._save_initial(self.output_dir, self.dataset_unique_id, self.unique_id, self.additional_tag_list,self.force_dir_replacement)
        
        df_output = self.cohort.get_dataframe()
        # No errors expected
        df_err = error_dataframe if error_dataframe is not None else self.error_dataframe
        return df_output, df_err
    
    def _run_quality_control(self,HTMLReport:"HTMLReport",nicePlots=True,offline_mode=True):
        """
        Override QC report.
        """
        import os
        import json
        from gencap.utils.dataset import validate_required_files,validate_data_description,validate_data_content,validate_file_existence
        hrQC = HTMLReport 
        hrQC.addSection("Cohort Validation for the gencap framework.")
        
        path = os.path.join(self.output_dir, self.dataset_unique_id)
        hrQC.addContent(f"<p><b>Saved Path:</b> {path} </p>")
        try:
            if validate_required_files(path=path):
                hrQC.addContent("<p><b>Validation of Required Files:</b> Succeeded </p>")
        except Exception as e:
            print(e)
            hrQC.addContent("<p><b>Validation of Required Files:</b> Failed </p>")
            
        try:
            if validate_data_description(path=path):
                hrQC.addContent("<p><b>Validation of Data Description:</b> Succeeded </p>")
        except:
            hrQC.addContent("<p><b>Validation of Data Description:</b> Failed </p>")
        
        try:
            if validate_data_content(path=path):
                hrQC.addContent("<p><b>Validation of Data Content:</b> Succeeded </p>")
        except:
            hrQC.addContent("<p><b>Validation of Data Content:</b> Failed </p>")
            
        try:
            data_desc = json.load(open(os.path.join(path, "data_source.json"), "r", encoding="utf-8"))
            if validate_file_existence(dataframe=self.output_dataframe,data_desc=data_desc):
                hrQC.addContent("<p><b>Validation of File Existence:</b> Succeeded </p>")
        except Exception as e:
            hrQC.addContent("<p><b>Validation of File Existence:</b> Failed </p>")
            hrQC.addContent(f"<li>{e}</li>")

        return hrQC

@task_manager.register("GENCAP-DatasetDisplayTask", version="1.0")
class DatasetDisplayTask(Task):
    def __init__(self):
        super().__init__()
        # Unique ID and version
        self.unique_id = "GENCAP-DatasetDisplayTask"
        self.version = "1.0"

        # Catalogue metadata
        self.application_domain = "Generic"
        self.application = ["utility", "display"]
        self.developer_names = "GENCAP Dev Team"
        self.developer_organization = "Inria"
        # Human-readable description
        self.description = (
            "This task creates a catalogue of datasets while providing their "
            "relationships and their links with pipelines and tasks."
        )
        # Parameters: root directory to scan for datasets
        self.parameter_info = {
            "database_dir": "(str) Root directory under which raw datasets and derived datasets are stored."
        }

        # No inputs required: we scan filesystem directly
        self.additional_input_indexing_fields = []
        self.mandatory_input_fields = []
        self.optional_input_fields = []
        self.input_file_location_fields = []

        # Outputs: one row per dataset, indexed by Dataset ID
        self.additional_output_indexing_fields = []
        self.mandatory_output_fields = ["Description","FilePath"]
        self.optional_output_fields = []
        self.output_file_location_fields = ["FilePath"]
        
       
    def _run_task(
        self,
        input_dataframe=None,
        output_dataframe=None,
        error_dataframe=None,
        base_indexing_fields=None,
        parameter_dict=None,
        pathname_policy_dict=None,
        debug_mode: bool = False
    ):
        if input_dataframe is None:
            input_dataframe = self.input_dataframe
        if output_dataframe is None:
            output_dataframe = self.output_dataframe
        if error_dataframe is None:
            error_dataframe = self.error_dataframe
        if base_indexing_fields is None:
            base_indexing_fields = self.base_indexing_fields
        if parameter_dict is None and self.parameter_dict is not None:
            parameter_dict = self.parameter_dict
        if pathname_policy_dict is None and self.pathname_policy_dict is not None:
            pathname_policy_dict = self.pathname_policy_dict
            
        import os
        from pathlib import Path
        import json
        import pandas as pd
        import gencap.utils.htmlReport as hr
        from gencap.utils.helpers import build_dependency_graph_dataset, visualize_graph_dataset, format_size

        # determine root to scan
        root = parameter_dict["database_dir"]
        
        records = []
        # walk through all subdirectories
        for dirpath, dirnames, filenames in os.walk(root):
            if "derivatives" in dirpath:
                data_type = "derived"
            elif "raw_data" in dirpath:
                data_type = "raw"
            elif "source_data" in dirpath:
                data_type = "source"
            else:
                data_type = "unknown"
            
            if "intermediate_data" in dirpath:
                inter_data = True
                parent_dataset= os.path.basename(os.path.dirname(dirpath.split('intermediate_data')[0]))
            else:
                inter_data = False
                parent_dataset= None
            
            size_dir = sum(f.lstat().st_size for f in Path(dirpath).rglob('*') if f.is_file())
                
            if "data_description.json" in filenames : # derivatives
                ds_id = os.path.basename(dirpath)
                # load metadata
                dd = json.load(open(os.path.join(dirpath, "data_description.json"), "r", encoding="utf-8"))
                if "data_source.json" in os.listdir(dirpath):
                    ds = json.load(open(os.path.join(dirpath, "data_source.json"), "r", encoding="utf-8"))
                if ds_id != self.dataset_unique_id:
                    records.append({
                        "Dataset_ID": ds_id,
                        "Dataset_Type": data_type,
                        "Dataset_Size": format_size(size_dir),
                        "Is_Intermediate_Dataset": inter_data,
                        "Parent_Dataset":parent_dataset,
                        "Path": dirpath,
                        "Tags": dd.get("tagList", []),
                        "Pipeline_ID": ds.get("pipelineID", None) if 'ds' in locals() else None,
                        "Pipeline_Version": ds.get("pipelineVersion", None) if 'ds' in locals() else None,
                        "Tasks": ",".join(ds.get("tasks", {}).keys()) if 'ds' in locals() else "",
                        "Cohort_Dataset_List": ds.get("cohortDatasetList", []) if 'ds' in locals() and ds["cohortDatasetList"] else [],
                        
                    })
        
        # build output dataframe
        self.base_indexing_fields = ["Dataset_ID"]
        df = pd.DataFrame(records)
        
        # also add "Has_Intermediate_Dataset" column
        parent_list = df['Parent_Dataset'].unique()
        df['Has_Intermediate_Dataset'] = df['Dataset_ID'].isin(parent_list)
        
        df_output_path = os.path.join(self.root_dir,"dataset_information.csv")
        df.to_csv(df_output_path)
        
        # Create an HTMLReport object for building the report
        hrRep = hr.HTMLReport(filename      = os.path.join(self.root_dir, "dataset_catalogue.html"),
                                title         = "Catalogue of the Dataset",
                                directory     = self.root_dir,
                                nicePlots     = True,
                                offline_mode  = False)
        
        hrRep.addSection("Data Catalogue")
        hrRep.addContent(f"<p><b>Database Path:</b> {self.parameter_dict['database_dir']} </p>")
        total_dataframe = df.reset_index()
        parent_list = list(total_dataframe['Parent_Dataset'].unique())
        dataset_links_html = "<p><b>Raw Data List:</b><ul>"
        raw_dataframe = total_dataframe[total_dataframe['Dataset_Type']=='raw']
        for i, row in raw_dataframe.iterrows():
            info_path = os.path.join(row['Path'],"information.html")
            if os.path.exists(info_path):
                try:
                    if "JUPYTERHUB_USER" in os.environ:
                        info_path = info_path.replace(os.environ["HOME"], os.path.join("/user",os.environ["JUPYTERHUB_USER"],"files"))
                    dataset_links_html += f"<li><a href='{info_path}' target='_top' rel='noopener noreferrer'>{row['Dataset_ID']}</a></li>"
                except Exception:
                    dataset_links_html += f"<li>{row['Dataset_ID']}</li>"
            else:
                dataset_links_html += f"<li>{row['Dataset_ID']}</li>"
        dataset_links_html +="</ul>"
        
        hrRep.addContent(dataset_links_html)
        dataset_links_html = "<p><b>Derived Data List:</b><ul>"
        derived_dataframe = total_dataframe[(total_dataframe['Dataset_Type']=="derived") & (total_dataframe['Is_Intermediate_Dataset']==False)]
        for i, row in derived_dataframe.iterrows():
            info_path = os.path.join(row['Path'],"information.html")
            if os.path.exists(info_path):
                try:
                    if "JUPYTERHUB_USER" in os.environ:
                        info_path = info_path.replace(os.environ["HOME"], os.path.join("/user",os.environ["JUPYTERHUB_USER"],"files"))
                    dataset_links_html += f"<li><a href='{info_path}' target='_top' rel='noopener noreferrer'>{row['Dataset_ID']}</a>"

                except Exception:
                    dataset_links_html += f"<li>{row['Dataset_ID']}"
            else:
                dataset_links_html += f"<li>{row['Dataset_ID']}"
            dataset_links_html += "<ul>"
            if row['Dataset_ID'] in parent_list:
                sub_df = total_dataframe.groupby('Parent_Dataset').get_group(row['Dataset_ID'])
                sub_df = sub_df.sort_values(by='Dataset_ID')
                for i, sub_row in sub_df.iterrows():
                    info_path = os.path.join(sub_row['Path'],"information.html")
                    if os.path.exists(info_path):
                        try:
                            if "JUPYTERHUB_USER" in os.environ:
                                info_path = info_path.replace(os.environ["HOME"], os.path.join("/user",os.environ["JUPYTERHUB_USER"],"files"))
                            dataset_links_html += f"<li><a href='{info_path}' target='_top' rel='noopener noreferrer'>{sub_row['Dataset_ID']}</a></li>"

                        except Exception:
                            dataset_links_html += f"<li>{sub_row['Dataset_ID']}</li>"
                    else:
                        dataset_links_html += f"<li>{sub_row['Dataset_ID']}</li>"
            dataset_links_html += "</ul></li>"
        dataset_links_html +="</ul>"
        hrRep.addContent(dataset_links_html)
        
        hrRep.addSubSection("Data Catalogue by Tag")
        tag_list = total_dataframe['Tags'].explode().dropna().unique().tolist()     
        tag_html = "<p><b>Tag List:</b><ul>"
        for tag in tag_list:
            tag_html += f"<li>{tag}</li>"
        tag_html +="</ul>"
        hrRep.addContent(tag_html)
        hrRep.addDataFrameFilter(
            df=total_dataframe,
            name="Dataset Catalogue with Tag Filters",
            filter_columns=["Tags"]
        )
        
        hrRep.addSubSection("Data Catalogue by Pipeline")
        pipe_list = total_dataframe["Pipeline_ID"].dropna().astype(str).unique()  
        pipe_html = "<p><b>Pipeline ID List:</b><ul>"
        for pipe in pipe_list:
            pipe_html += f"<li>{pipe}</li>"
        pipe_html +="</ul>"
        hrRep.addContent(pipe_html)
        hrRep.addDataFrameFilter(
            df=total_dataframe,
            name="Dataset Catalogue with Pipeline Filters",
            filter_columns=["Pipeline_ID"]
        )
        
        hrRep.saveToHTML()
        
        # build output_dataframe, error_dataframe
        self.base_indexing_fields = ["FileName"]
        error_dataframe = self.error_dataframe
        data = [{"FileName": "dataset_information.csv",
                 "Description": "A dataframe which has the information about the datasets located in database_dir ",
                 "FilePath": df_output_path},
                {"FileName": "dataset_catalogue.html",
                 "Description": "A catalogue of datasets while providing theirrelationships and their links with pipelines and tasks.",
                 "FilePath": os.path.join(self.root_dir, "dataset_catalogue.html")}]
        output_dataframe = pd.DataFrame(data)
        return output_dataframe, error_dataframe

        # No inputs required: we scan filesystem directly

        self.mandatory_input_fields = []
        self.optional_input_fields = []
        self.input_file_location_fields = []

        # Outputs: one row per dataset, indexed by dataset ID
        self.additional_output_indexing_fields = []
        self.mandatory_output_fields =  ["Dataset Path","Dataset Size", "Checked At", "Validation Status",

                                         "Num Checks", "Num Failures", "Failed Checks","Failed Reasons","Num Rows", "Num Columns"]
        self.optional_output_fields = []

    def _run_task(
        self,
        input_dataframe: pd.DataFrame = None,
        output_dataframe: pd.DataFrame = None,
        error_dataframe: pd.DataFrame = None,
        base_indexing_fields: list = None,
        parameter_dict: dict = None,
        pathname_policy_dict: dict = None,
        debug_mode: bool = False,
    ):
 

        
        import traceback
        import os
        from pathlib import Path
        import pandas as pd
        from gencap.utils.helpers import format_size
        from gencap.utils.dataset import validate_required_files,validate_data_description,validate_data_content,validate_file_existence

        import json
        # Update the dataset_path to list:
        if isinstance(self.parameter_dict["dataset_path"], str):
            self.parameter_dict["dataset_path"] = [self.parameter_dict["dataset_path"]]
        
        records = []
        # Iterate the path to validate
        for dataset_path in self.parameter_dict["dataset_path"]:
            # initialize contatiner
            row_dict={}
            fail_desc= []
            # convert the path to prevent misinterpretation
            if dataset_path.endswith('/'):
                dataset_path = os.path.dirname(dataset_path)
                
            # check RequiredFiles
            try:
                row_dict['Valid_RequiredFiles'] = validate_required_files(path=dataset_path)
                self.logger.info(f"{os.path.basename(dataset_path)} - {dataset_path} succeed to pass [RequiredFiles] validation process.")
            except Exception as e:
                row_dict['Valid_RequiredFiles'] = False
                # Log any exceptions encountered during processing
                error_row = {"Base Indexing Value": os.path.basename(dataset_path), #dataset ID
                            "Additional Indexing Value": "",
                            "Error Code": "Valid_RequiredFiles Fail",
                            "Description": str(type(e).__name__)+':'+str(e),
                            "Values": str(traceback.format_exc())}
                self.error_dataframe = pd.concat([self.error_dataframe, pd.DataFrame([error_row])], ignore_index=True)
                self.logger.error(f"{error_row['Base Indexing Value']} - {dataset_path} Failed to pass [RequiredFiles] validation process.")
                self.logger.error(f"{error_row['Base Indexing Value']} - {error_row['Description']}")
                self.logger.debug(f"{error_row['Base Indexing Value']} - Traceback message:")
                self.logger.debug(f"{error_row['Base Indexing Value']} - {error_row['Values']}")
                fail_desc.append(error_row['Description'])
            # check DataDescription
            try:
                row_dict['Valid_DataDescription'] = validate_data_description(path=dataset_path)
                self.logger.info(f"{os.path.basename(dataset_path)} - {dataset_path} succeed to pass [DataDescription] validation process.")
            except Exception as e:
                row_dict['Valid_DataDescription'] = False
                # Log any exceptions encountered during processing
                error_row = {"Base Indexing Value": os.path.basename(dataset_path), #dataset ID
                            "Additional Indexing Value": "",
                            "Error Code": "Valid_DataDescription Fail",
                            "Description": str(type(e).__name__)+':'+str(e),
                            "Values": str(traceback.format_exc())}
                self.error_dataframe = pd.concat([self.error_dataframe, pd.DataFrame([error_row])], ignore_index=True)
                self.logger.error(f"{error_row['Base Indexing Value']} - {dataset_path} Failed to pass [DataDescription] validation process.")
                self.logger.error(f"{error_row['Base Indexing Value']} - {error_row['Description']}")
                self.logger.debug(f"{error_row['Base Indexing Value']} - Traceback message:")
                self.logger.debug(f"{error_row['Base Indexing Value']} - {error_row['Values']}")
                fail_desc.append(error_row['Description'])
                
            # check DataContent
            try:
                row_dict['Valid_DataContent'] = validate_data_content(path=dataset_path)
                self.logger.info(f"{os.path.basename(dataset_path)} - {dataset_path} succeed to pass [DataContent] validation process.")
            except Exception as e:
                row_dict['Valid_DataContent'] = False
                # Log any exceptions encountered during processing
                error_row = {"Base Indexing Value": os.path.basename(dataset_path), #dataset ID
                            "Additional Indexing Value": "",
                            "Error Code": "Valid_DataContent Fail",
                            "Description": str(type(e).__name__)+':'+str(e),
                            "Values": str(traceback.format_exc())}
                self.error_dataframe = pd.concat([self.error_dataframe, pd.DataFrame([error_row])], ignore_index=True)
                self.logger.error(f"{error_row['Base Indexing Value']} - {dataset_path} Failed to pass [DataContent] validation process.")
                self.logger.error(f"{error_row['Base Indexing Value']} - {error_row['Description']}")
                self.logger.debug(f"{error_row['Base Indexing Value']} - Traceback message:")
                self.logger.debug(f"{error_row['Base Indexing Value']} - {error_row['Values']}")
                fail_desc.append(error_row['Description'])
                
            # check FileExistence
            try:
                row_dict['Valid_FileExistence'] = validate_file_existence(path=dataset_path)
                self.logger.info(f"{os.path.basename(dataset_path)} - {dataset_path} succeed to pass [FileExistence] validation process.")
            except Exception as e:
                row_dict['Valid_FileExistence'] = False
                # Log any exceptions encountered during processing
                error_row = {"Base Indexing Value": os.path.basename(dataset_path), #dataset ID
                            "Additional Indexing Value": "",
                            "Error Code": "Valid_FileExistence Fail",
                            "Description": str(type(e).__name__)+':'+str(e),
                            "Values": str(traceback.format_exc())}
                self.error_dataframe = pd.concat([self.error_dataframe, pd.DataFrame([error_row])], ignore_index=True)
                self.logger.error(f"{error_row['Base Indexing Value']} - {dataset_path} Failed to pass [FileExistence] validation process.")
                self.logger.error(f"{error_row['Base Indexing Value']} - {error_row['Description']}")
                self.logger.debug(f"{error_row['Base Indexing Value']} - Traceback message:")
                self.logger.debug(f"{error_row['Base Indexing Value']} - {error_row['Values']}")
                fail_desc.append(error_row['Description'])
                
            # Collect additional metrics
            try:
                dataframe = pd.read_pickle(os.path.join(dataset_path,"data_content.pkl"))
                n_row = dataframe.shape[0]
                n_col = dataframe.shape[1]
            except Exception as e:
                n_row = None
                n_col = None
            status = "Failed" if not all(row_dict.values()) else "Succeeded"
            num_checks = len(row_dict)
            num_fail = sum(1 for v in row_dict.values() if v is False)
            failed_checks = [k for k,v in row_dict.items() if v is False]
            size_dir = sum(f.lstat().st_size for f in Path(dataset_path).rglob('*') if f.is_file())
            # Consist other columns
            row_dict["Dataset ID"] = os.path.basename(dataset_path)
            row_dict["Dataset Path"] = dataset_path
            row_dict["Dataset Size"] = format_size(size_dir)
            row_dict["Checked At"] = pd.Timestamp.now()
            row_dict["Validation Status"] = status
            row_dict["Num Checks"] = num_checks
            row_dict["Num Failures"] = num_fail
            row_dict["Failed Checks"] = failed_checks
            row_dict["Failed Reasons"] = fail_desc if len(fail_desc)>0 else None
            row_dict["Num Rows"] = n_row
            row_dict["Num Columns"] = n_col
            
            
            records.append(row_dict)
            
        # build output dataframe
        self.base_indexing_fields = ["Dataset ID"]
        df = pd.DataFrame(records)
        error_dataframe = self.error_dataframe

        else:
            output_dataframe = df.set_index(self.base_indexing_fields)
        
        return output_dataframe, error_dataframe
